112     public boolean start() {

115         }
116         //initialization of the game
117         currRound.randomStartingPlayer(playersList);
118         bag.initialBagFill();

120         islandsList = Island.generateIslandsList();
121         Island.initStudentIsland(islandsList, motherNaturePos, bag);
122 
123         bag.addAllStudents();
124         cloudsList = Cloud.generateCloudsList(numOfPlayers);
125         currRound.fillClouds(cloudsList, bag, numOfPlayers);
126 
127         for (PawnColor c : PawnColor.values()) {
128             Professor p = new Professor(c);
129             tableProfessorsList.add(p);
130         }
131 
132         for (Player p : playersList) {
133             p.getDashboard().placeStudentEntrance(numOfPlayers, bag);
134             p.getDashboard().generateTower(numOfPlayers, p.getTowerColor());
135         }
136 
137         if (expertMode) {
138             charactersList = Character.extractCharacters();
139             tableMoney = new AtomicInteger();
140             tableMoney.set(20);
141             for (Player p : playersList) {
142                 p.modifyMoney(1, tableMoney);
143             }
144             CharacterParameters parameters = new CharacterParameters();
145             parameters.setBag(bag);
146 
147             for (Character c : charactersList) {
148                 c.initCharacter(parameters);
149                 //the method getForbidCharacter search if there is id 5 otherwise return null
150             }
151         }
152         state = GameState.INGAME_STATE;
153         sendInitGame();
154         sendInGameState(); //give information about current state and prints the game
155         return true;
156     }
