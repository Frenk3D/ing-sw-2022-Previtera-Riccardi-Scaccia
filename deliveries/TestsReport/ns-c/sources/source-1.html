97                 scanner.reset();
98                 if (address.equals("") || address.equals(" ") || address.equals("\n")) {
99                     serverInfo.put("address", defaultAddress);

102                     serverInfo.put("address", address);
103                     validInput = true;
104                 } else {

106                     clearCli();
107                     validInput = false;
108                 }
109             } while (!validInput);
110

114                 scanner.reset();
115 
116                 if (port.equals("") || port.equals(" ") || port.equals("\n")) {
117                     serverInfo.put("port", defaultPort);

121                         serverInfo.put("port", port);

124                         out.println("Invalid port!");
