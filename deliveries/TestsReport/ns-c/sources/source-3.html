309      * Writes text at a given position in a matrix
310      *
311      * @param matrix the matrix
312      * @param x      the position in the x axis
313      * @param y      the position in the y axis
314      * @param text   the text to be written
315      */
316     private void writeAtPos(String[] matrix, int x, int y, String text) {
317         String[] result = matrix;
318         int ansiLengthText = 0;
319         int ansiLenghtMatrix = 0;
320         Pattern pattern = Pattern.compile("\u001B\\[.*?m");
321         Matcher matcher = pattern.matcher(text);
322         while (matcher.find()) {
323             String s = matcher.group(0);
324             ansiLengthText += s.length();
325         }
326 
327         matcher = pattern.matcher(matrix[y]);
328         while (matcher.find()) {
329             int len = matcher.group(0).length();
330             if (matcher.start() < x + len + ansiLenghtMatrix) {
331                 ansiLenghtMatrix += len;
332             }
333         }
334 
335         StringBuilder stringBuilder = new StringBuilder(result[y]);
336         // replace substring from index 5 to index 9
337         stringBuilder.replace(x + ansiLenghtMatrix, x + ansiLenghtMatrix + text.length() - ansiLengthText, text);
338         result[y] = stringBuilder.toString();
339     }
